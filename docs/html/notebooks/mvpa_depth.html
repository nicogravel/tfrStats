import os
import numpy as np
from tfrMVPA.plot_mvtfr_stats import plot_mvtfr_stats as plot_mvtfr_stats


fbands  = ['low','high','higher','mua']
blocks   = ['grat','obj','bck','nat']


# Loop over blocks and frequency bands
for  block in range(4):
    for  fband in range(3):

        # Load data
        fname = os.path.join('/Volumes/Gandalf/MWlamprop/results/stats/multivariate/mvtfr_'
                    + blocks[block] + '_'
                    + fbands[fband]
                    + '.npy')
        print(fname)
        tfrs = np.load(fname,allow_pickle=True).item()
        tfr = tfrs['tfr']
        tfr_null = tfrs['tfr_null']
        plot_mvtfr_stats(tfr, tfr_null, fband, clust_corr = False, cluster_size = 3, alpha = 0.05)
